\newlength{\TempTableLength}

\newenvironment{TableEnvironment}{
  \vskip 5mm%
  \begin{table}[!t]%
  \begin{center}%
}{
  \end{center}%
  \end{table}%
}

\renewcommand{\thetable}{\Roman{table}}

\newcommand{\Table}[5]{%
  \begin{TableEnvironment}%
    \refstepcounter{table}%
    \addcontentsline{lot}{section}{TABLE \thetable{}~~~#1}
    \label{tab:#5}%
    TABLE \thetable{}\\[1ex]%
    \settowidth{\TempTableLength}{\textsc{#1}}%
    \ifdim \TempTableLength > 0.95\columnwidth%
      \parbox{0.95\columnwidth}{\textsc{#1}}%
    \else%
      \textsc{#1}%
    \fi\\[1em]%
    \renewcommand{\arraystretch}{1.2}%
    \begin{tabular}{#2}%
      \hline%
      #3\\%
      \hline%
      #4%
      \hline%
    \end{tabular}
  \end{TableEnvironment}%
}
